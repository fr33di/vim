########## Comandos para vim #########
:w "Guardar"
:q "Salir"
:q! "Salir sin guardar"
:wq "Guardar y salir"
:qa "Cerrar ventanas y termina vim"
:qa! "Fuerza la salida de vim"
u "Deshacer acción o ctrl z"
Ctrl+r "Rehacer una acción"
gg "Ponerse al inicio"
Mayús+g "Ir a la última línea"
2G "Ir a la linea 2 en el modo normal"
:set number "Mostrar lineas"
:set nonumber "Ocultar lineas"
$ "Ir al final de la líne"
0 "Ir al principio de la línea"
dd "Borrar linea actual"
d+num "Borrar un número de líneas"
/+texto "Función de búsqueda (n y N)"
:h "Abre la ayuda principal"
1,2d "Para cortar las lineas 1 a 2"
3,4y "Para copiar las lineas 1a 2"
p "Para pegar lo cortado o copiado despues del cursor"
P "Para pegar lo cortado o copiado antes del cursor"
